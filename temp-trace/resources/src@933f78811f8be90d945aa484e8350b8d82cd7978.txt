/**
 * @fileoverview Address validation tests for the checkout flow.
 * Tests form validation for shipping and billing addresses,
 * covering valid, invalid, and edge-case inputs.
 * 
 * Tags: @regression, @checkout, @validation
 */

import { test, expect } from '../../src/fixtures/test.fixtures';
import { CheckoutPage } from '../../src/pages/checkout.page';
import { BasketPage } from '../../src/pages/basket.page';
import { HomePage } from '../../src/pages/home.page';
import { ProductDetailPage } from '../../src/pages/product-detail.page';
import { ProductListingPage } from '../../src/pages/product-listing.page';
import { TestData } from '../../src/helpers/test-data';
import { TestConfig } from '../../src/config/test.config';

test.describe('Address Validation @checkout @validation', () => {
  /**
   * Setup: Add product to cart and navigate to checkout address step.
   * Returns the checkout page, or null if checkout is inaccessible.
   */
  async function navigateToAddressStep(
    page: import('@playwright/test').Page
  ): Promise<CheckoutPage | null> {
    const homePage = new HomePage(page);
    const plp = new ProductListingPage(page);
    const pdp = new ProductDetailPage(page);
    const basketPage = new BasketPage(page);
    const checkoutPage = new CheckoutPage(page);

    await homePage.open();
    await homePage.searchAndSubmit('t-shirt');
    await plp.clickFirstProduct();
    await pdp.selectFirstAvailableSize();
    await pdp.addToBasket();
    await basketPage.open();
    await basketPage.proceedToCheckout();

    // Check if we reached the address form
    const firstNameVisible = await checkoutPage.isElementVisible(
      checkoutPage.firstNameInput, 10000
    );

    return firstNameVisible ? checkoutPage : null;
  }

  // ─── Positive Tests ─────────────────────────────────────────

  test.describe('Valid Addresses @regression', () => {

    test('Provide valid address and verify that checkout continues @regression', async ({ page, homePage, productListingPage, productDetailPage, basketPage, checkoutPage, loginPage }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      // 1. First login
      await test.step('Login with AboutYou credentials', async () => {
        await homePage.open();
        // Click account icon to open login modal
        const accountIcon = page.getByRole('button', { name: /Login|Account/i }).first();
        await accountIcon.waitFor({ state: 'visible' });
        await accountIcon.click();
        
        await loginPage.loginForm.waitFor({ state: 'visible', timeout: 10000 });
        
        // This will now prompt the user if a CAPTCHA is detected
        await loginPage.login(TestConfig.credentials.email, TestConfig.credentials.password);

        await page.waitForTimeout(3000); // Wait for successful login

        // Store the state as requested (this ensures the current context uses this state)
        await page.context().storageState({ path: TestConfig.storageStatePath });
      });

      // 2. Add items to basket as logged in user
      await test.step('Add product to basket', async () => {
        await homePage.open();
        await homePage.searchAndSubmit('t-shirt');
        await productListingPage.clickFirstProduct();
        await productDetailPage.selectFirstAvailableSize();
        await productDetailPage.addToBasket();
        await page.keyboard.press('Escape'); // close mini-basket
        await page.waitForTimeout(1000);
      });

      // 3. Proceed to checkout
      await basketPage.open();
      await basketPage.proceedToCheckout();

      // Check if we reached the address form
      const firstNameVisible = await checkoutPage.isElementVisible(
        checkoutPage.firstNameInput, 10000
      );

      if (!firstNameVisible) {
         test.skip(true, 'Could not reach address step');
         return;
      }

      await checkoutPage.fillShippingAddress(TestData.VALID_DE_ADDRESS);
      await checkoutPage.clickContinue();

      // Should not show validation errors
      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(false);
    });


    test('should accept address with special characters (umlauts) @regression', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.SPECIAL_CHARS_ADDRESS);
      await checkoutPage.clickContinue();

      // Umlauts and special characters should be accepted
      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(false);
    });

    test('should accept Austrian address @regression', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.VALID_AT_ADDRESS);
      await checkoutPage.clickContinue();

      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(false);
    });
  });

  // ─── Negative Tests ─────────────────────────────────────────

  test.describe('Invalid Addresses @negative', () => {

    test('should show errors for empty required fields @regression @negative', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      // Submit without filling any fields
      await checkoutPage.clickContinue();

      // Should show validation errors
      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(true);
    });

    test('should reject invalid postal code format @regression @negative', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.INVALID_POSTCODE_ADDRESS);
      await checkoutPage.clickContinue();

      // Should show validation error for postal code
      const errors = await checkoutPage.getValidationErrors();
      // At least one error should be present
      expect(errors.length).toBeGreaterThan(0);
    });

    test('should reject invalid email format @regression @negative', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.INVALID_EMAIL_ADDRESS);
      await checkoutPage.clickContinue();

      // Should show validation error for email
      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(true);
    });
  });

  // ─── Edge Cases ─────────────────────────────────────────────

  test.describe('Address Edge Cases @edge-case', () => {

    test('should handle extremely long input values @edge-case', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.LONG_VALUES_ADDRESS);
      await checkoutPage.clickContinue();

      // Should either accept (if valid) or show validation error — not crash
      const pageTitle = await page.title();
      expect(pageTitle).toBeTruthy(); // Page is still functional
    });

    test('should reject whitespace-only input @edge-case', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.WHITESPACE_ADDRESS);
      await checkoutPage.clickContinue();

      // Whitespace-only should be treated as empty — show errors
      const hasErrors = await checkoutPage.hasValidationErrors();
      expect(hasErrors).toBe(true);
    });

    test('should accept minimal valid input @edge-case', async ({ page }) => {
      test.skip(!TestConfig.hasCredentials(), 'No test credentials configured');

      const checkoutPage = await navigateToAddressStep(page);
      if (!checkoutPage) {
        test.skip(true, 'Could not reach address step');
        return;
      }

      await checkoutPage.fillShippingAddress(TestData.MINIMAL_ADDRESS);
      await checkoutPage.clickContinue();

      // Single-character names should be accepted
      const hasErrors = await checkoutPage.hasValidationErrors();
      // This may or may not pass depending on minimum length requirements
    });
  });

  // ─── Cleanup ────────────────────────────────────────────────

  test.afterEach(async ({ page }) => {
    try {
      const basket = new BasketPage(page);
      await basket.open();
      await basket.clearCart();
    } catch {
      // Best-effort cleanup
    }
  });
});
